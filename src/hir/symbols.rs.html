25/// The actual data that is stored in the index. It should be as compact as
26/// possible.
27#[derive(Debug, Clone, PartialEq, Eq, Hash)]
28pub struct FileSymbol {
29    pub name: Symbol,
30    pub def: ModuleDef,
31    pub loc: DeclarationLocation,
32    pub container_name: Option<Symbol>,
33    /// Whether this symbol is a doc alias for the original symbol.
34    pub is_alias: bool,
35    pub is_assoc: bool,
36    pub is_import: bool,
37    pub do_not_complete: Complete,
38}
39
40#[derive(Debug, Clone, PartialEq, Eq, Hash)]
41pub struct DeclarationLocation {
42    /// The file id for both the `ptr` and `name_ptr`.
43    pub hir_file_id: HirFileId,
44    /// This points to the whole syntax node of the declaration.
45    pub ptr: SyntaxNodePtr,
46    /// This points to the [`syntax::ast::Name`] identifier of the declaration.
47    pub name_ptr: AstPtr<Either<syntax::ast::Name, syntax::ast::NameRef>>,
48}
49
50impl DeclarationLocation {
51    pub fn syntax<DB: HirDatabase>(&self, sema: &Semantics<'_, DB>) -> SyntaxNode {
52        let root = sema.parse_or_expand(self.hir_file_id);
53        self.ptr.to_node(&root)
54    }
55}

57/// Represents an outstanding module that the symbol collector must collect symbols from.
58#[derive(Debug)]
59struct SymbolCollectorWork {
60    module_id: ModuleId,
61    parent: Option<Name>,
62}
63
64pub struct SymbolCollector<'a> {
65    db: &'a dyn HirDatabase,
66    symbols: FxIndexSet<FileSymbol>,
67    work: Vec<SymbolCollectorWork>,
68    current_container_name: Option<Symbol>,
69    collect_pub_only: bool,
70}
71
72/// Given a [`ModuleId`] and a [`HirDatabase`], use the DefMap for the module's crate to collect
73/// all symbols that should be indexed for the given module.
74impl<'a> SymbolCollector<'a> {
75    pub fn new(db: &'a dyn HirDatabase, collect_pub_only: bool) -> Self {
76        SymbolCollector {
77            db,
78            symbols: Default::default(),
79            work: Default::default(),
80            current_container_name: None,
81            collect_pub_only,
82        }
83    }
84
85    pub fn new_module(
86        db: &dyn HirDatabase,
87        module: Module,
88        collect_pub_only: bool,
89    ) -> Box<[FileSymbol]> {
90        let mut symbol_collector = SymbolCollector::new(db, collect_pub_only);
91        symbol_collector.collect(module);
92        symbol_collector.finish()
93    }
94
95    pub fn collect(&mut self, module: Module) {
96        let _p = tracing::info_span!("SymbolCollector::collect", ?module).entered();
97        tracing::info!(?module, "SymbolCollector::collect");
98
99        // The initial work is the root module we're collecting, additional work will
100        // be populated as we traverse the module's definitions.
101        self.work.push(SymbolCollectorWork { module_id: module.into(), parent: None });
102
103        while let Some(work) = self.work.pop() {
104            self.do_work(work);
105        }
106    }
107
108    pub fn finish(self) -> Box<[FileSymbol]> {
109        self.symbols.into_iter().collect()
110    }
111
112    fn do_work(&mut self, work: SymbolCollectorWork) {
113        let _p = tracing::info_span!("SymbolCollector::do_work", ?work).entered();
114        tracing::info!(?work, "SymbolCollector::do_work");
115        self.db.unwind_if_revision_cancelled();
116
117        let parent_name = work.parent.map(|name| Symbol::intern(name.as_str()));
118        self.with_container_name(parent_name, |s| s.collect_from_module(work.module_id));
119    }
120
121    fn collect_from_module(&mut self, module_id: ModuleId) {
122        let collect_pub_only = self.collect_pub_only;
123        let push_decl = |this: &mut Self, def: ModuleDefId, name, vis| {
124            if collect_pub_only && vis != Visibility::Public {
125                return;
126            }
127            match def {
128                ModuleDefId::ModuleId(id) => this.push_module(id, name),
129                ModuleDefId::FunctionId(id) => {
130                    this.push_decl(id, name, false, None);
131                    this.collect_from_body(id, Some(name.clone()));
132                }
133                ModuleDefId::AdtId(AdtId::StructId(id)) => {
134                    this.push_decl(id, name, false, None);
135                }
136                ModuleDefId::AdtId(AdtId::EnumId(id)) => {
137                    this.push_decl(id, name, false, None);
138                    let enum_name = Symbol::intern(this.db.enum_signature(id).name.as_str());
139                    this.with_container_name(Some(enum_name), |this| {
140                        let variants = id.enum_variants(this.db);
141                        for (variant_id, variant_name, _) in &variants.variants {
142                            this.push_decl(*variant_id, variant_name, true, None);
143                        }
144                    });
145                }
146                ModuleDefId::AdtId(AdtId::UnionId(id)) => {
147                    this.push_decl(id, name, false, None);
148                }
149                ModuleDefId::ConstId(id) => {
150                    this.push_decl(id, name, false, None);
151                    this.collect_from_body(id, Some(name.clone()));
152                }
153                ModuleDefId::StaticId(id) => {
154                    this.push_decl(id, name, false, None);
155                    this.collect_from_body(id, Some(name.clone()));
156                }
157                ModuleDefId::TraitId(id) => {
158                    let trait_do_not_complete = this.push_decl(id, name, false, None);
159                    this.collect_from_trait(id, trait_do_not_complete);
160                }
161                ModuleDefId::TypeAliasId(id) => {
162                    this.push_decl(id, name, false, None);
163                }
164                ModuleDefId::MacroId(id) => {
165                    match id {
166                        MacroId::Macro2Id(id) => this.push_decl(id, name, false, None),
167                        MacroId::MacroRulesId(id) => this.push_decl(id, name, false, None),
168                        MacroId::ProcMacroId(id) => this.push_decl(id, name, false, None),
169                    };
170                }
171                // Don't index these.
172                ModuleDefId::BuiltinType(_) => {}
173                ModuleDefId::EnumVariantId(_) => {}
174            }
175        };
176
177        // Nested trees are very common, so a cache here will hit a lot.
178        let import_child_source_cache = &mut FxHashMap::default();
179
180        let is_explicit_import = |vis| match vis {
181            Visibility::Public => true,
182            Visibility::PubCrate(_) => true,
183            Visibility::Module(_, VisibilityExplicitness::Explicit) => true,
184            Visibility::Module(_, VisibilityExplicitness::Implicit) => false,
185        };
186
187        let mut push_import = |this: &mut Self, i: ImportId, name: &Name, def: ModuleDefId, vis| {
188            if collect_pub_only && vis != Visibility::Public {
189                return;
190            }
191            let source = import_child_source_cache
192                .entry(i.use_)
193                .or_insert_with(|| i.use_.child_source(this.db));
194            let Some(use_tree_src) = source.value.get(i.idx) else { return };
195            let rename = use_tree_src.rename().and_then(|rename| rename.name());
196            let name_syntax = match rename {
197                Some(name) => Some(Either::Left(name)),
198                None if is_explicit_import(vis) => {
199                    (|| use_tree_src.path()?.segment()?.name_ref().map(Either::Right))()
200                }
201                None => None,
202            };
203            let Some(name_syntax) = name_syntax else {
204                return;
205            };
206            let dec_loc = DeclarationLocation {
207                hir_file_id: source.file_id,
208                ptr: SyntaxNodePtr::new(use_tree_src.syntax()),
209                name_ptr: AstPtr::new(&name_syntax),
210            };
211            this.symbols.insert(FileSymbol {
212                name: name.symbol().clone(),
213                def: def.into(),
214                container_name: this.current_container_name.clone(),
215                loc: dec_loc,
216                is_alias: false,
217                is_assoc: false,
218                is_import: true,
219                do_not_complete: Complete::Yes,
220            });
221        };
222
223        let push_extern_crate =
224            |this: &mut Self, i: ExternCrateId, name: &Name, def: ModuleDefId, vis| {
225                if collect_pub_only && vis != Visibility::Public {
226                    return;
227                }
228                let loc = i.lookup(this.db);
229                let source = loc.source(this.db);
230                let rename = source.value.rename().and_then(|rename| rename.name());
231
232                let name_syntax = match rename {
233                    Some(name) => Some(Either::Left(name)),
234                    None if is_explicit_import(vis) => None,
235                    None => source.value.name_ref().map(Either::Right),
236                };
237                let Some(name_syntax) = name_syntax else {
238                    return;
239                };
240                let dec_loc = DeclarationLocation {
241                    hir_file_id: source.file_id,
242                    ptr: SyntaxNodePtr::new(source.value.syntax()),
243                    name_ptr: AstPtr::new(&name_syntax),
244                };
245                this.symbols.insert(FileSymbol {
246                    name: name.symbol().clone(),
247                    def: def.into(),
248                    container_name: this.current_container_name.clone(),
249                    loc: dec_loc,
250                    is_alias: false,
251                    is_assoc: false,
252                    is_import: false,
253                    do_not_complete: Complete::Yes,
254                });
255            };
256
257        let def_map = module_id.def_map(self.db);
258        let scope = &def_map[module_id.local_id].scope;
259
260        for impl_id in scope.impls() {
261            self.collect_from_impl(impl_id);
262        }
263
264        for (name, Item { def, vis, import }) in scope.types() {
265            if let Some(i) = import {
266                match i {
267                    ImportOrExternCrate::Import(i) => push_import(self, i, name, def, vis),
268                    ImportOrExternCrate::Glob(_) => (),
269                    ImportOrExternCrate::ExternCrate(i) => {
270                        push_extern_crate(self, i, name, def, vis)
271                    }
272                }
273
274                continue;
275            }
276            // self is a declaration
277            push_decl(self, def, name, vis)
278        }
279
280        for (name, Item { def, vis, import }) in scope.macros() {
281            if let Some(i) = import {
282                match i {
283                    ImportOrExternCrate::Import(i) => push_import(self, i, name, def.into(), vis),
284                    ImportOrExternCrate::Glob(_) => (),
285                    ImportOrExternCrate::ExternCrate(_) => (),
286                }
287                continue;
288            }
289            // self is a declaration
290            push_decl(self, ModuleDefId::MacroId(def), name, vis)
291        }
292
293        for (name, Item { def, vis, import }) in scope.values() {
294            if let Some(i) = import {
295                match i {
296                    ImportOrGlob::Import(i) => push_import(self, i, name, def, vis),
297                    ImportOrGlob::Glob(_) => (),
298                }
299                continue;
300            }
301            // self is a declaration
302            push_decl(self, def, name, vis)
303        }
304
305        for const_id in scope.unnamed_consts() {
306            self.collect_from_body(const_id, None);
307        }
308
309        for (name, id) in scope.legacy_macros() {
310            for &id in id {
311                if id.module(self.db) == module_id {
312                    match id {
313                        MacroId::Macro2Id(id) => self.push_decl(id, name, false, None),
314                        MacroId::MacroRulesId(id) => self.push_decl(id, name, false, None),
315                        MacroId::ProcMacroId(id) => self.push_decl(id, name, false, None),
316                    };
317                }
318            }
319        }
320    }

322    fn collect_from_body(&mut self, body_id: impl Into<DefWithBodyId>, name: Option<Name>) {
323        if self.collect_pub_only {
324            return;
325        }
326        let body_id = body_id.into();
327        let body = self.db.body(body_id);
328
329        // Descend into the blocks and enqueue collection of all modules within.
330        for (_, def_map) in body.blocks(self.db) {
331            for (id, _) in def_map.modules() {
332                self.work.push(SymbolCollectorWork {
333                    module_id: def_map.module_id(id),
334                    parent: name.clone(),
335                });
336            }
337        }
338    }

340    fn collect_from_impl(&mut self, impl_id: ImplId) {
341        let impl_data = self.db.impl_signature(impl_id);
342        let impl_name = Some(
343            hir_display_with_store(impl_data.self_ty, &impl_data.store)
344                .display(
345                    self.db,
346                    crate::Impl::from(impl_id).krate(self.db).to_display_target(self.db),
347                )
348                .to_smolstr(),
349        );
350        self.with_container_name(impl_name.as_deref().map(Symbol::intern), |s| {
351            for &(ref name, assoc_item_id) in &impl_id.impl_items(self.db).items {
352                if s.collect_pub_only && s.db.assoc_visibility(assoc_item_id) != Visibility::Public
353                {
354                    continue;
355                }
356
357                s.push_assoc_item(assoc_item_id, name, None)
358            }
359        })
360    }
361
362    fn collect_from_trait(&mut self, trait_id: TraitId, trait_do_not_complete: Complete) {
363        let trait_data = self.db.trait_signature(trait_id);
364        self.with_container_name(Some(Symbol::intern(trait_data.name.as_str())), |s| {
365            for &(ref name, assoc_item_id) in &trait_id.trait_items(self.db).items {
366                s.push_assoc_item(assoc_item_id, name, Some(trait_do_not_complete));
367            }
368        });
369    }
370
371    fn with_container_name(&mut self, container_name: Option<Symbol>, f: impl FnOnce(&mut Self)) {
372        if let Some(container_name) = container_name {
373            let prev = self.current_container_name.replace(container_name);
374            f(self);
375            self.current_container_name = prev;
376        } else {
377            f(self);
378        }
379    }
380
381    fn push_assoc_item(
382        &mut self,
383        assoc_item_id: AssocItemId,
384        name: &Name,
385        trait_do_not_complete: Option<Complete>,
386    ) {
387        match assoc_item_id {
388            AssocItemId::FunctionId(id) => self.push_decl(id, name, true, trait_do_not_complete),
389            AssocItemId::ConstId(id) => self.push_decl(id, name, true, trait_do_not_complete),
390            AssocItemId::TypeAliasId(id) => self.push_decl(id, name, true, trait_do_not_complete),
391        };
392    }
393
394    fn push_decl<L>(
395        &mut self,
396        id: L,
397        name: &Name,
398        is_assoc: bool,
399        trait_do_not_complete: Option<Complete>,
400    ) -> Complete
401    where
402        L: Lookup<Database = dyn DefDatabase> + Into<ModuleDefId>,
403        <L as Lookup>::Data: HasSource,
404        <<L as Lookup>::Data as HasSource>::Value: HasName,
405    {
406        let loc = id.lookup(self.db);
407        let source = loc.source(self.db);
408        let Some(name_node) = source.value.name() else { return Complete::Yes };
409        let def = ModuleDef::from(id.into());
410        let dec_loc = DeclarationLocation {
411            hir_file_id: source.file_id,
412            ptr: SyntaxNodePtr::new(source.value.syntax()),
413            name_ptr: AstPtr::new(&name_node).wrap_left(),
414        };
415
416        let mut do_not_complete = Complete::Yes;
417
418        if let Some(attrs) = def.attrs(self.db) {
419            do_not_complete = Complete::extract(matches!(def, ModuleDef::Trait(_)), &attrs);
420            if let Some(trait_do_not_complete) = trait_do_not_complete {
421                do_not_complete = Complete::for_trait_item(trait_do_not_complete, do_not_complete);
422            }
423
424            for alias in attrs.doc_aliases() {
425                self.symbols.insert(FileSymbol {
426                    name: alias.clone(),
427                    def,
428                    loc: dec_loc.clone(),
429                    container_name: self.current_container_name.clone(),
430                    is_alias: true,
431                    is_assoc,
432                    is_import: false,
433                    do_not_complete,
434                });
435            }
436        }
437
438        self.symbols.insert(FileSymbol {
439            name: name.symbol().clone(),
440            def,
441            container_name: self.current_container_name.clone(),
442            loc: dec_loc,
443            is_alias: false,
444            is_assoc,
445            is_import: false,
446            do_not_complete,
447        });
448
449        do_not_complete
450    }

452    fn push_module(&mut self, module_id: ModuleId, name: &Name) {
453        let def_map = module_id.def_map(self.db);
454        let module_data = &def_map[module_id.local_id];
455        let Some(declaration) = module_data.origin.declaration() else { return };
456        let module = declaration.to_node(self.db);
457        let Some(name_node) = module.name() else { return };
458        let dec_loc = DeclarationLocation {
459            hir_file_id: declaration.file_id,
460            ptr: SyntaxNodePtr::new(module.syntax()),
461            name_ptr: AstPtr::new(&name_node).wrap_left(),
462        };
463
464        let def = ModuleDef::Module(module_id.into());
465
466        let mut do_not_complete = Complete::Yes;
467        if let Some(attrs) = def.attrs(self.db) {
468            do_not_complete = Complete::extract(matches!(def, ModuleDef::Trait(_)), &attrs);
469
470            for alias in attrs.doc_aliases() {
471                self.symbols.insert(FileSymbol {
472                    name: alias.clone(),
473                    def,
474                    loc: dec_loc.clone(),
475                    container_name: self.current_container_name.clone(),
476                    is_alias: true,
477                    is_assoc: false,
478                    is_import: false,
479                    do_not_complete,
480                });
481            }
482        }
483
484        self.symbols.insert(FileSymbol {
485            name: name.symbol().clone(),
486            def: ModuleDef::Module(module_id.into()),
487            container_name: self.current_container_name.clone(),
488            loc: dec_loc,
489            is_alias: false,
490            is_assoc: false,
491            is_import: false,
492            do_not_complete,
493        });
494    }
