1//! Implementation of "param name" inlay hints:
2//! ```no_run
3//! fn max(x: i32, y: i32) -> i32 { x + y }
4//! _ = max(/*x*/4, /*y*/4);
5//! ```

18pub(super) fn hints(
19    acc: &mut Vec<InlayHint>,
20    FamousDefs(sema, krate): &FamousDefs<'_, '_>,
21    config: &InlayHintsConfig<'_>,
22    file_id: EditionedFileId,
23    expr: ast::Expr,
24) -> Option<()> {
25    if !config.parameter_hints {
26        return None;
27    }
28
29    let (callable, arg_list) = get_callable(sema, &expr)?;
30    let unary_function = callable.n_params() == 1;
31    let function_name = match callable.kind() {
32        hir::CallableKind::Function(function) => Some(function.name(sema.db)),
33        _ => None,
34    };
35    let function_name = function_name.as_ref().map(|it| it.as_str());
36    let hints = callable
37        .params()
38        .into_iter()
39        .zip(arg_list.args())
40        .filter_map(|(p, arg)| {
41            // Only annotate hints for expressions that exist in the original file
42            let range = sema.original_range_opt(arg.syntax())?;
43            if range.file_id != file_id {
44                return None;
45            }
46            let param_name = p.name(sema.db)?;
47            Some((p, param_name, arg, range))
48        })
49        .filter(|(_, param_name, arg, _)| {
50            !should_hide_param_name_hint(
51                sema,
52                unary_function,
53                function_name,
54                param_name.as_str(),
55                arg,
56            )
57        })
58        .map(|(param, param_name, _, hir::FileRange { range, .. })| {
59            let colon = if config.render_colons { ":" } else { "" };
60            let label = InlayHintLabel::simple(
61                format!("{}{colon}", param_name.display(sema.db, krate.edition(sema.db))),
62                None,
63                config.lazy_location_opt(|| {
64                    let source = sema.source(param)?;
65                    let name_syntax = match source.value.as_ref() {
66                        Either::Left(pat) => pat.name(),
67                        Either::Right(param) => match param.pat()? {
68                            ast::Pat::IdentPat(it) => it.name(),
69                            _ => None,
70                        },
71                    }?;
72                    sema.original_range_opt(name_syntax.syntax()).map(|frange| ide_db::FileRange {
73                        file_id: frange.file_id.file_id(sema.db),
74                        range: frange.range,
75                    })
76                }),
77            );
78            InlayHint {
79                range,
80                kind: InlayKind::Parameter,
81                label,
82                text_edit: None,
83                position: InlayHintPosition::Before,
84                pad_left: false,
85                pad_right: true,
86                resolve_parent: Some(expr.syntax().text_range()),
87            }
88        });
89
90    acc.extend(hints);
91    Some(())
92}

94fn get_callable<'db>(
95    sema: &Semantics<'db, RootDatabase>,
96    expr: &ast::Expr,
97) -> Option<(hir::Callable<'db>, ast::ArgList)> {
98    match expr {
99        ast::Expr::CallExpr(expr) => {
100            let descended = sema.descend_node_into_attributes(expr.clone()).pop();
101            let expr = descended.as_ref().unwrap_or(expr);
102            sema.type_of_expr(&expr.expr()?)?.original.as_callable(sema.db).zip(expr.arg_list())
103        }
104        ast::Expr::MethodCallExpr(expr) => {
105            let descended = sema.descend_node_into_attributes(expr.clone()).pop();
106            let expr = descended.as_ref().unwrap_or(expr);
107            sema.resolve_method_call_as_callable(expr).zip(expr.arg_list())
108        }
109        _ => None,
110    }
111}

113const INSIGNIFICANT_METHOD_NAMES: &[&str] = &["clone", "as_ref", "into"];
114const INSIGNIFICANT_PARAMETER_NAMES: &[&str] =
115    &["predicate", "value", "pat", "rhs", "other", "msg", "op"];

117fn should_hide_param_name_hint(
118    sema: &Semantics<'_, RootDatabase>,
119    unary_function: bool,
120    function_name: Option<&str>,
121    param_name: &str,
122    argument: &ast::Expr,
123) -> bool {
124    // These are to be tested in the `parameter_hint_heuristics` test
125    // hide when:
126    // - the parameter name is a suffix of the function's name
127    // - the argument is a qualified constructing or call expression where the qualifier is an ADT
128    // - exact argument<->parameter match(ignoring leading and trailing underscore) or
129    //   parameter is a prefix/suffix of argument with _ splitting it off
130    // - param starts with `ra_fixture`
131    // - param is a well known name in a unary function
132
133    let param_name = param_name.trim_matches('_');
134    if param_name.is_empty() {
135        return true;
136    }
137
138    if param_name.starts_with("ra_fixture") {
139        return true;
140    }
141
142    if unary_function {
143        if let Some(function_name) = function_name
144            && is_param_name_suffix_of_fn_name(param_name, function_name)
145        {
146            return true;
147        }
148        if is_obvious_param(param_name) {
149            return true;
150        }
151    }
152
153    is_argument_expr_similar_to_param_name(sema, argument, param_name)
154}

156/// Hide the parameter name of a unary function if it is a `_` - prefixed suffix of the function's name, or equal.
157///
158/// `fn strip_suffix(suffix)` will be hidden.
159/// `fn stripsuffix(suffix)` will not be hidden.
160fn is_param_name_suffix_of_fn_name(param_name: &str, fn_name: &str) -> bool {
161    fn_name == param_name
162        || fn_name
163            .len()
164            .checked_sub(param_name.len())
165            .and_then(|at| fn_name.is_char_boundary(at).then(|| fn_name.split_at(at)))
166            .is_some_and(|(prefix, suffix)| {
167                suffix.eq_ignore_ascii_case(param_name) && prefix.ends_with('_')
168            })
169}

171fn is_argument_expr_similar_to_param_name(
172    sema: &Semantics<'_, RootDatabase>,
173    argument: &ast::Expr,
174    param_name: &str,
175) -> bool {
176    match get_segment_representation(argument) {
177        Some(Either::Left(argument)) => is_argument_similar_to_param_name(&argument, param_name),
178        Some(Either::Right(path)) => {
179            path.segment()
180                .and_then(|it| it.name_ref())
181                .is_some_and(|name_ref| name_ref.text().eq_ignore_ascii_case(param_name))
182                || is_adt_constructor_similar_to_param_name(sema, &path, param_name)
183        }
184        None => false,
185    }
186}

188/// Check whether param_name and argument are the same or
189/// whether param_name is a prefix/suffix of argument(split at `_`).
190pub(super) fn is_argument_similar_to_param_name(
191    argument: &[ast::NameRef],
192    param_name: &str,
193) -> bool {
194    debug_assert!(!argument.is_empty());
195    debug_assert!(!param_name.is_empty());
196    let param_name = param_name.split('_');
197    let argument = argument.iter().flat_map(|it| it.text_non_mutable().split('_'));
198
199    let prefix_match = zip(argument.clone(), param_name.clone())
200        .all(|(arg, param)| arg.eq_ignore_ascii_case(param));
201    let postfix_match = || {
202        zip(argument.rev(), param_name.rev()).all(|(arg, param)| arg.eq_ignore_ascii_case(param))
203    };
204    prefix_match || postfix_match()
205}

207pub(super) fn get_segment_representation(
208    expr: &ast::Expr,
209) -> Option<Either<Vec<ast::NameRef>, ast::Path>> {
210    match expr {
211        ast::Expr::MethodCallExpr(method_call_expr) => {
212            let receiver =
213                method_call_expr.receiver().and_then(|expr| get_segment_representation(&expr));
214            let name_ref = method_call_expr.name_ref()?;
215            if INSIGNIFICANT_METHOD_NAMES.contains(&name_ref.text().as_str()) {
216                return receiver;
217            }
218            Some(Either::Left(match receiver {
219                Some(Either::Left(mut left)) => {
220                    left.push(name_ref);
221                    left
222                }
223                Some(Either::Right(_)) | None => vec![name_ref],
224            }))
225        }
226        ast::Expr::FieldExpr(field_expr) => {
227            let expr = field_expr.expr().and_then(|expr| get_segment_representation(&expr));
228            let name_ref = field_expr.name_ref()?;
229            let res = match expr {
230                Some(Either::Left(mut left)) => {
231                    left.push(name_ref);
232                    left
233                }
234                Some(Either::Right(_)) | None => vec![name_ref],
235            };
236            Some(Either::Left(res))
237        }
238        // paths
239        ast::Expr::MacroExpr(macro_expr) => macro_expr.macro_call()?.path().map(Either::Right),
240        ast::Expr::RecordExpr(record_expr) => record_expr.path().map(Either::Right),
241        ast::Expr::PathExpr(path_expr) => {
242            let path = path_expr.path()?;
243            // single segment paths are likely locals
244            Some(match path.as_single_name_ref() {
245                None => Either::Right(path),
246                Some(name_ref) => Either::Left(vec![name_ref]),
247            })
248        }
249        ast::Expr::PrefixExpr(prefix_expr) if prefix_expr.op_kind() == Some(UnaryOp::Not) => None,
250        // recurse
251        ast::Expr::PrefixExpr(prefix_expr) => get_segment_representation(&prefix_expr.expr()?),
252        ast::Expr::RefExpr(ref_expr) => get_segment_representation(&ref_expr.expr()?),
253        ast::Expr::CastExpr(cast_expr) => get_segment_representation(&cast_expr.expr()?),
254        ast::Expr::CallExpr(call_expr) => get_segment_representation(&call_expr.expr()?),
255        ast::Expr::AwaitExpr(await_expr) => get_segment_representation(&await_expr.expr()?),
256        ast::Expr::IndexExpr(index_expr) => get_segment_representation(&index_expr.base()?),
257        ast::Expr::ParenExpr(paren_expr) => get_segment_representation(&paren_expr.expr()?),
258        ast::Expr::TryExpr(try_expr) => get_segment_representation(&try_expr.expr()?),
259        // ast::Expr::ClosureExpr(closure_expr) => todo!(),
260        _ => None,
261    }
262}

264fn is_obvious_param(param_name: &str) -> bool {
265    // avoid displaying hints for common functions like map, filter, etc.
266    // or other obvious words used in std
267    param_name.len() == 1 || INSIGNIFICANT_PARAMETER_NAMES.contains(&param_name)
268}

270fn is_adt_constructor_similar_to_param_name(
271    sema: &Semantics<'_, RootDatabase>,
272    path: &ast::Path,
273    param_name: &str,
274) -> bool {
275    (|| match sema.resolve_path(path)? {
276        hir::PathResolution::Def(hir::ModuleDef::Adt(_)) => {
277            Some(to_lower_snake_case(&path.segment()?.name_ref()?.text()) == param_name)
278        }
279        hir::PathResolution::Def(hir::ModuleDef::Function(_) | hir::ModuleDef::Variant(_)) => {
280            if to_lower_snake_case(&path.segment()?.name_ref()?.text()) == param_name {
281                return Some(true);
282            }
283            let qual = path.qualifier()?;
284            match sema.resolve_path(&qual)? {
285                hir::PathResolution::Def(hir::ModuleDef::Adt(_)) => {
286                    Some(to_lower_snake_case(&qual.segment()?.name_ref()?.text()) == param_name)
287                }
288                _ => None,
289            }
290        }
291        _ => None,
292    })()
293    .unwrap_or(false)
294}
