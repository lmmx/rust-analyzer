111#[derive(Debug)]
112pub enum MethodError<'db> {
113    /// Did not find an applicable method.
114    NoMatch,
115
116    /// Multiple methods might apply.
117    Ambiguity(Vec<CandidateSource>),
118
119    /// Found an applicable method, but it is not visible.
120    PrivateMatch(Pick<'db>),
121
122    /// Found a `Self: Sized` bound where `Self` is a trait object.
123    IllegalSizedBound { candidates: Vec<FunctionId>, needs_mut: bool },
124
125    /// Error has already been emitted, no need to emit another one.
126    ErrorReported,
127}

137impl<'a, 'db> InferenceContext<'a, 'db> {
138    /// Performs method lookup. If lookup is successful, it will return the callee
139    /// and store an appropriate adjustment for the self-expr. In some cases it may
140    /// report an error (e.g., invoking the `drop` method).
141    #[instrument(level = "debug", skip(self))]
142    pub(crate) fn lookup_method_including_private(
143        &mut self,
144        self_ty: Ty<'db>,
145        name: Name,
146        generic_args: Option<&HirGenericArgs>,
147        receiver: ExprId,
148        call_expr: ExprId,
149    ) -> Result<(MethodCallee<'db>, bool), MethodError<'db>> {
150        let (pick, is_visible) = match self.lookup_probe(name, self_ty) {
151            Ok(it) => (it, true),
152            Err(MethodError::PrivateMatch(it)) => {
153                // FIXME: Report error.
154                (it, false)
155            }
156            Err(err) => return Err(err),
157        };
158
159        let result = self.confirm_method(&pick, self_ty, call_expr, generic_args);
160        debug!("result = {:?}", result);
161
162        if result.illegal_sized_bound {
163            // FIXME: Report an error.
164        }
165
166        self.write_expr_adj(receiver, result.adjustments);
167        self.write_method_resolution(call_expr, result.callee.def_id, result.callee.args);
168
169        Ok((result.callee, is_visible))
170    }
171
172    #[instrument(level = "debug", skip(self))]
173    pub(crate) fn lookup_probe(
174        &self,
175        method_name: Name,
176        self_ty: Ty<'db>,
177    ) -> probe::PickResult<'db> {
178        self.with_method_resolution(|ctx| {
179            let pick = ctx.probe_for_name(probe::Mode::MethodCall, method_name, self_ty)?;
180            Ok(pick)
181        })
182    }
183
184    pub(crate) fn with_method_resolution<R>(
185        &self,
186        f: impl FnOnce(&MethodResolutionContext<'_, 'db>) -> R,
187    ) -> R {
188        let traits_in_scope = self.get_traits_in_scope();
189        let traits_in_scope = match &traits_in_scope {
190            Either::Left(it) => it,
191            Either::Right(it) => *it,
192        };
193        let ctx = MethodResolutionContext {
194            infcx: &self.table.infer_ctxt,
195            resolver: &self.resolver,
196            env: &self.table.trait_env,
197            traits_in_scope,
198            edition: self.edition,
199            unstable_features: &self.unstable_features,
200        };
201        f(&ctx)
202    }
203}

205/// Used by [FnCtxt::lookup_method_for_operator] with `-Znext-solver`.
206///
207/// With `AsRigid` we error on `impl Opaque: NotInItemBounds` while
208/// `AsInfer` just treats it as ambiguous and succeeds. This is necessary
209/// as we want [FnCtxt::check_expr_call] to treat not-yet-defined opaque
210/// types as rigid to support `impl Deref<Target = impl FnOnce()>` and
211/// `Box<impl FnOnce()>`.
212///
213/// We only want to treat opaque types as rigid if we need to eagerly choose
214/// between multiple candidates. We otherwise treat them as ordinary inference
215/// variable to avoid rejecting otherwise correct code.
216#[derive(Debug)]
217#[expect(dead_code)]
218pub(super) enum TreatNotYetDefinedOpaques {
219    AsInfer,
220    AsRigid,
221}

223impl<'db> InferenceTable<'db> {
224    /// `lookup_method_in_trait` is used for overloaded operators.
225    /// It does a very narrow slice of what the normal probe/confirm path does.
226    /// In particular, it doesn't really do any probing: it simply constructs
227    /// an obligation for a particular trait with the given self type and checks
228    /// whether that trait is implemented.
229    #[instrument(level = "debug", skip(self))]
230    pub(super) fn lookup_method_for_operator(
231        &self,
232        cause: ObligationCause,
233        method_name: Symbol,
234        trait_def_id: TraitId,
235        self_ty: Ty<'db>,
236        opt_rhs_ty: Option<Ty<'db>>,
237        treat_opaques: TreatNotYetDefinedOpaques,
238    ) -> Option<InferOk<'db, MethodCallee<'db>>> {
239        // Construct a trait-reference `self_ty : Trait<input_tys>`
240        let args = GenericArgs::for_item(
241            self.interner(),
242            trait_def_id.into(),
243            |param_idx, param_id, _| match param_id {
244                GenericParamId::LifetimeParamId(_) | GenericParamId::ConstParamId(_) => {
245                    unreachable!("did not expect operator trait to have lifetime/const")
246                }
247                GenericParamId::TypeParamId(_) => {
248                    if param_idx == 0 {
249                        self_ty.into()
250                    } else if let Some(rhs_ty) = opt_rhs_ty {
251                        assert_eq!(param_idx, 1, "did not expect >1 param on operator trait");
252                        rhs_ty.into()
253                    } else {
254                        // FIXME: We should stop passing `None` for the failure case
255                        // when probing for call exprs. I.e. `opt_rhs_ty` should always
256                        // be set when it needs to be.
257                        self.next_var_for_param(param_id)
258                    }
259                }
260            },
261        );
262
263        let obligation = Obligation::new(
264            self.interner(),
265            cause,
266            self.trait_env.env,
267            TraitRef::new_from_args(self.interner(), trait_def_id.into(), args),
268        );
269
270        // Now we want to know if this can be matched
271        let matches_trait = match treat_opaques {
272            TreatNotYetDefinedOpaques::AsInfer => self.infer_ctxt.predicate_may_hold(&obligation),
273            TreatNotYetDefinedOpaques::AsRigid => {
274                self.infer_ctxt.predicate_may_hold_opaque_types_jank(&obligation)
275            }
276        };
277
278        if !matches_trait {
279            debug!("--> Cannot match obligation");
280            // Cannot be matched, no such method resolution is possible.
281            return None;
282        }
283
284        // Trait must have a method named `m_name` and it should not have
285        // type parameters or early-bound regions.
286        let interner = self.interner();
287        // We use `Ident::with_dummy_span` since no built-in operator methods have
288        // any macro-specific hygiene, so the span's context doesn't really matter.
289        let Some(method_item) =
290            trait_def_id.trait_items(self.db).method_by_name(&Name::new_symbol_root(method_name))
291        else {
292            panic!("expected associated item for operator trait")
293        };
294
295        let def_id = method_item;
296
297        debug!("lookup_in_trait_adjusted: method_item={:?}", method_item);
298        let mut obligations = PredicateObligations::new();
299
300        // Instantiate late-bound regions and instantiate the trait
301        // parameters into the method type to get the actual method type.
302        //
303        // N.B., instantiate late-bound regions before normalizing the
304        // function signature so that normalization does not need to deal
305        // with bound regions.
306        let fn_sig =
307            self.db.callable_item_signature(method_item.into()).instantiate(interner, args);
308        let fn_sig = self
309            .infer_ctxt
310            .instantiate_binder_with_fresh_vars(BoundRegionConversionTime::FnCall, fn_sig);
311
312        // Register obligations for the parameters. This will include the
313        // `Self` parameter, which in turn has a bound of the main trait,
314        // so this also effectively registers `obligation` as well. (We
315        // used to register `obligation` explicitly, but that resulted in
316        // double error messages being reported.)
317        //
318        // Note that as the method comes from a trait, it should not have
319        // any late-bound regions appearing in its bounds.
320        let bounds = GenericPredicates::query_all(self.db, method_item.into());
321        let bounds = clauses_as_obligations(
322            bounds.iter_instantiated_copied(interner, args.as_slice()),
323            ObligationCause::new(),
324            self.trait_env.env,
325        );
326
327        obligations.extend(bounds);
328
329        // Also add an obligation for the method type being well-formed.
330        debug!(
331            "lookup_method_in_trait: matched method fn_sig={:?} obligation={:?}",
332            fn_sig, obligation
333        );
334        for ty in fn_sig.inputs_and_output {
335            obligations.push(Obligation::new(
336                interner,
337                obligation.cause.clone(),
338                self.trait_env.env,
339                Binder::dummy(PredicateKind::Clause(ClauseKind::WellFormed(ty.into()))),
340            ));
341        }
342
343        let callee = MethodCallee { def_id, args, sig: fn_sig };
344        debug!("callee = {:?}", callee);
345
346        Some(InferOk { obligations, value: callee })
347    }
348}

350pub fn lookup_impl_const<'db>(
351    infcx: &InferCtxt<'db>,
352    env: Arc<TraitEnvironment<'db>>,
353    const_id: ConstId,
354    subs: GenericArgs<'db>,
355) -> (ConstId, GenericArgs<'db>) {
356    let interner = infcx.interner;
357    let db = interner.db;
358
359    let trait_id = match const_id.loc(db).container {
360        ItemContainerId::TraitId(id) => id,
361        _ => return (const_id, subs),
362    };
363    let trait_ref = TraitRef::new(interner, trait_id.into(), subs);
364
365    let const_signature = db.const_signature(const_id);
366    let name = match const_signature.name.as_ref() {
367        Some(name) => name,
368        None => return (const_id, subs),
369    };
370
371    lookup_impl_assoc_item_for_trait_ref(infcx, trait_ref, env, name)
372        .and_then(
373            |assoc| if let (AssocItemId::ConstId(id), s) = assoc { Some((id, s)) } else { None },
374        )
375        .unwrap_or((const_id, subs))
376}

378/// Checks if the self parameter of `Trait` method is the `dyn Trait` and we should
379/// call the method using the vtable.
380pub fn is_dyn_method<'db>(
381    interner: DbInterner<'db>,
382    _env: Arc<TraitEnvironment<'db>>,
383    func: FunctionId,
384    fn_subst: GenericArgs<'db>,
385) -> Option<usize> {
386    let db = interner.db;
387
388    let ItemContainerId::TraitId(trait_id) = func.loc(db).container else {
389        return None;
390    };
391    let trait_params = db.generic_params(trait_id.into()).len();
392    let fn_params = fn_subst.len() - trait_params;
393    let trait_ref = TraitRef::new(
394        interner,
395        trait_id.into(),
396        GenericArgs::new_from_iter(interner, fn_subst.iter().take(trait_params)),
397    );
398    let self_ty = trait_ref.self_ty();
399    if let TyKind::Dynamic(d, _) = self_ty.kind() {
400        // rustc doesn't accept `impl Foo<2> for dyn Foo<5>`, so if the trait id is equal, no matter
401        // what the generics are, we are sure that the method is come from the vtable.
402        let is_my_trait_in_bounds = d
403            .principal_def_id()
404            .is_some_and(|trait_| all_super_traits(db, trait_.0).contains(&trait_id));
405        if is_my_trait_in_bounds {
406            return Some(fn_params);
407        }
408    }
409    None
410}

412/// Looks up the impl method that actually runs for the trait method `func`.
413///
414/// Returns `func` if it's not a method defined in a trait or the lookup failed.
415pub(crate) fn lookup_impl_method_query<'db>(
416    db: &'db dyn HirDatabase,
417    env: Arc<TraitEnvironment<'db>>,
418    func: FunctionId,
419    fn_subst: GenericArgs<'db>,
420) -> (FunctionId, GenericArgs<'db>) {
421    let interner = DbInterner::new_with(db, Some(env.krate), env.block);
422    let infcx = interner.infer_ctxt().build(TypingMode::PostAnalysis);
423
424    let ItemContainerId::TraitId(trait_id) = func.loc(db).container else {
425        return (func, fn_subst);
426    };
427    let trait_params = db.generic_params(trait_id.into()).len();
428    let trait_ref = TraitRef::new(
429        interner,
430        trait_id.into(),
431        GenericArgs::new_from_iter(interner, fn_subst.iter().take(trait_params)),
432    );
433
434    let name = &db.function_signature(func).name;
435    let Some((impl_fn, impl_subst)) =
436        lookup_impl_assoc_item_for_trait_ref(&infcx, trait_ref, env, name).and_then(|assoc| {
437            if let (AssocItemId::FunctionId(id), subst) = assoc { Some((id, subst)) } else { None }
438        })
439    else {
440        return (func, fn_subst);
441    };
442
443    (
444        impl_fn,
445        GenericArgs::new_from_iter(
446            interner,
447            impl_subst.iter().chain(fn_subst.iter().skip(trait_params)),
448        ),
449    )
450}
451
452fn lookup_impl_assoc_item_for_trait_ref<'db>(
453    infcx: &InferCtxt<'db>,
454    trait_ref: TraitRef<'db>,
455    env: Arc<TraitEnvironment<'db>>,
456    name: &Name,
457) -> Option<(AssocItemId, GenericArgs<'db>)> {
458    let (impl_id, impl_subst) = find_matching_impl(infcx, &env, trait_ref)?;
459    let item =
460        impl_id.impl_items(infcx.interner.db).items.iter().find_map(|(n, it)| match *it {
461            AssocItemId::FunctionId(f) => (n == name).then_some(AssocItemId::FunctionId(f)),
462            AssocItemId::ConstId(c) => (n == name).then_some(AssocItemId::ConstId(c)),
463            AssocItemId::TypeAliasId(_) => None,
464        })?;
465    Some((item, impl_subst))
466}

468pub(crate) fn find_matching_impl<'db>(
469    infcx: &InferCtxt<'db>,
470    env: &TraitEnvironment<'db>,
471    trait_ref: TraitRef<'db>,
472) -> Option<(ImplId, GenericArgs<'db>)> {
473    let trait_ref =
474        infcx.at(&ObligationCause::dummy(), env.env).deeply_normalize(trait_ref).ok()?;
475
476    let obligation = Obligation::new(infcx.interner, ObligationCause::dummy(), env.env, trait_ref);
477
478    let selection = infcx.select(&obligation).ok()??;
479
480    // Currently, we use a fulfillment context to completely resolve
481    // all nested obligations. This is because they can inform the
482    // inference of the impl's type parameters.
483    let mut ocx = ObligationCtxt::new(infcx);
484    let impl_source = selection.map(|obligation| ocx.register_obligation(obligation));
485
486    let errors = ocx.evaluate_obligations_error_on_ambiguity();
487    if !errors.is_empty() {
488        return None;
489    }
490
491    let impl_source = infcx.resolve_vars_if_possible(impl_source);
492    if impl_source.has_non_region_infer() {
493        return None;
494    }
495
496    match impl_source {
497        ImplSource::UserDefined(impl_source) => Some((impl_source.impl_def_id, impl_source.args)),
498        ImplSource::Param(_) | ImplSource::Builtin(..) => None,
499    }
500}

509pub fn incoherent_inherent_impls(db: &dyn HirDatabase, self_ty: SimplifiedType) -> &[ImplId] {
510    let has_incoherent_impls = match self_ty.def() {
511        Some(def_id) => match def_id.try_into() {
512            Ok(def_id) => {
513                db.attrs(def_id).by_key(sym::rustc_has_incoherent_inherent_impls).exists()
514            }
515            Err(()) => true,
516        },
517        _ => true,
518    };
519    return if !has_incoherent_impls {
520        &[]
521    } else {
522        incoherent_inherent_impls_query(db, (), self_ty)
523    };
524
525    #[salsa::tracked(returns(ref))]
526    fn incoherent_inherent_impls_query(
527        db: &dyn HirDatabase,
528        _force_query_input_to_be_interned: (),
529        self_ty: SimplifiedType,
530    ) -> Box<[ImplId]> {
531        let _p = tracing::info_span!("incoherent_inherent_impl_crates").entered();
532
533        let mut result = Vec::new();
534        for &krate in crates_containing_incoherent_inherent_impls(db) {
535            let impls = InherentImpls::for_crate(db, krate);
536            result.extend_from_slice(impls.for_self_ty(&self_ty));
537        }
538        result.into_boxed_slice()
539    }
540}

577impl InherentImpls {
578    fn collect_def_map(db: &dyn HirDatabase, def_map: &DefMap) -> Self {
579        let mut map = FxHashMap::default();
580        collect(db, def_map, &mut map);
581        let mut map = map
582            .into_iter()
583            .map(|(self_ty, impls)| (self_ty, impls.into_boxed_slice()))
584            .collect::<FxHashMap<_, _>>();
585        map.shrink_to_fit();
586        return Self { map };
587
588        fn collect(
589            db: &dyn HirDatabase,
590            def_map: &DefMap,
591            map: &mut FxHashMap<SimplifiedType, Vec<ImplId>>,
592        ) {
593            for (_module_id, module_data) in def_map.modules() {
594                for impl_id in module_data.scope.impls() {
595                    let data = db.impl_signature(impl_id);
596                    if data.target_trait.is_some() {
597                        continue;
598                    }
599
600                    let interner = DbInterner::new_with(db, None, None);
601                    let self_ty = db.impl_self_ty(impl_id);
602                    let self_ty = self_ty.instantiate_identity();
603                    if let Some(self_ty) =
604                        simplify_type(interner, self_ty, TreatParams::InstantiateWithInfer)
605                    {
606                        map.entry(self_ty).or_default().push(impl_id);
607                    }
608                }
609
610                // To better support custom derives, collect impls in all unnamed const items.
611                // const _: () = { ... };
612                for konst in module_data.scope.unnamed_consts() {
613                    let body = db.body(konst.into());
614                    for (_, block_def_map) in body.blocks(db) {
615                        collect(db, block_def_map, map);
616                    }
617                }
618            }
619        }
620    }
621
622    pub fn for_self_ty(&self, self_ty: &SimplifiedType) -> &[ImplId] {
623        self.map.get(self_ty).map(|it| &**it).unwrap_or_default()
624    }
625
626    pub fn for_each_crate_and_block(
627        db: &dyn HirDatabase,
628        krate: Crate,
629        block: Option<BlockId>,
630        for_each: &mut dyn FnMut(&InherentImpls),
631    ) {
632        let blocks = std::iter::successors(block, |block| block.loc(db).module.containing_block());
633        blocks.filter_map(|block| Self::for_block(db, block).as_deref()).for_each(&mut *for_each);
634        for_each(Self::for_crate(db, krate));
635    }
636}

638#[derive(Debug, PartialEq)]
639struct OneTraitImpls {
640    non_blanket_impls: FxHashMap<SimplifiedType, Box<[ImplId]>>,
641    blanket_impls: Box<[ImplId]>,
642}
643
644#[derive(Default)]
645struct OneTraitImplsBuilder {
646    non_blanket_impls: FxHashMap<SimplifiedType, Vec<ImplId>>,
647    blanket_impls: Vec<ImplId>,
648}
649
650impl OneTraitImplsBuilder {
651    fn finish(self) -> OneTraitImpls {
652        let mut non_blanket_impls = self
653            .non_blanket_impls
654            .into_iter()
655            .map(|(self_ty, impls)| (self_ty, impls.into_boxed_slice()))
656            .collect::<FxHashMap<_, _>>();
657        non_blanket_impls.shrink_to_fit();
658        let blanket_impls = self.blanket_impls.into_boxed_slice();
659        OneTraitImpls { non_blanket_impls, blanket_impls }
660    }
661}
662
663#[derive(Debug, PartialEq)]
664pub struct TraitImpls {
665    map: FxHashMap<TraitId, OneTraitImpls>,
666}
667
668#[salsa::tracked]
669impl TraitImpls {
670    #[salsa::tracked(returns(ref))]
671    pub fn for_crate(db: &dyn HirDatabase, krate: Crate) -> Arc<Self> {
672        let _p = tracing::info_span!("inherent_impls_in_crate_query", ?krate).entered();
673
674        let crate_def_map = crate_def_map(db, krate);
675        let result = Self::collect_def_map(db, crate_def_map);
676        Arc::new(result)
677    }
678
679    #[salsa::tracked(returns(ref))]
680    pub fn for_block(db: &dyn HirDatabase, block: BlockId) -> Option<Box<Self>> {
681        let _p = tracing::info_span!("inherent_impls_in_block_query").entered();
682
683        let block_def_map = block_def_map(db, block);
684        let result = Self::collect_def_map(db, block_def_map);
685        if result.map.is_empty() { None } else { Some(Box::new(result)) }
686    }
687
688    #[salsa::tracked(returns(ref))]
689    pub fn for_crate_and_deps(db: &dyn HirDatabase, krate: Crate) -> Box<[Arc<Self>]> {
690        db.transitive_deps(krate).iter().map(|&dep| Self::for_crate(db, dep).clone()).collect()
691    }
692}
693
694impl TraitImpls {
695    fn collect_def_map(db: &dyn HirDatabase, def_map: &DefMap) -> Self {
696        let mut map = FxHashMap::default();
697        collect(db, def_map, &mut map);
698        let mut map = map
699            .into_iter()
700            .map(|(trait_id, trait_map)| (trait_id, trait_map.finish()))
701            .collect::<FxHashMap<_, _>>();
702        map.shrink_to_fit();
703        return Self { map };
704
705        fn collect(
706            db: &dyn HirDatabase,
707            def_map: &DefMap,
708            map: &mut FxHashMap<TraitId, OneTraitImplsBuilder>,
709        ) {
710            for (_module_id, module_data) in def_map.modules() {
711                for impl_id in module_data.scope.impls() {
712                    // Reservation impls should be ignored during trait resolution, so we never need
713                    // them during type analysis. See rust-lang/rust#64631 for details.
714                    //
715                    // FIXME: Reservation impls should be considered during coherence checks. If we are
716                    // (ever) to implement coherence checks, this filtering should be done by the trait
717                    // solver.
718                    if db.attrs(impl_id.into()).by_key(sym::rustc_reservation_impl).exists() {
719                        continue;
720                    }
721                    let trait_ref = match db.impl_trait(impl_id) {
722                        Some(tr) => tr.instantiate_identity(),
723                        None => continue,
724                    };
725                    let self_ty = trait_ref.self_ty();
726                    let interner = DbInterner::new_with(db, None, None);
727                    let entry = map.entry(trait_ref.def_id.0).or_default();
728                    match simplify_type(interner, self_ty, TreatParams::InstantiateWithInfer) {
729                        Some(self_ty) => {
730                            entry.non_blanket_impls.entry(self_ty).or_default().push(impl_id)
731                        }
732                        None => entry.blanket_impls.push(impl_id),
733                    }
734                }
735
736                // To better support custom derives, collect impls in all unnamed const items.
737                // const _: () = { ... };
738                for konst in module_data.scope.unnamed_consts() {
739                    let body = db.body(konst.into());
740                    for (_, block_def_map) in body.blocks(db) {
741                        collect(db, block_def_map, map);
742                    }
743                }
744            }
745        }
746    }
747
748    pub fn blanket_impls(&self, for_trait: TraitId) -> &[ImplId] {
749        self.map.get(&for_trait).map(|it| &*it.blanket_impls).unwrap_or_default()
750    }
751
752    /// Queries whether `self_ty` has potentially applicable implementations of `trait_`.
753    pub fn has_impls_for_trait_and_self_ty(
754        &self,
755        trait_: TraitId,
756        self_ty: &SimplifiedType,
757    ) -> bool {
758        self.map.get(&trait_).is_some_and(|trait_impls| {
759            trait_impls.non_blanket_impls.contains_key(self_ty)
760                || !trait_impls.blanket_impls.is_empty()
761        })
762    }
763
764    pub fn for_trait_and_self_ty(&self, trait_: TraitId, self_ty: &SimplifiedType) -> &[ImplId] {
765        self.map
766            .get(&trait_)
767            .and_then(|map| map.non_blanket_impls.get(self_ty))
768            .map(|it| &**it)
769            .unwrap_or_default()
770    }
771
772    pub fn for_trait(&self, trait_: TraitId, mut callback: impl FnMut(&[ImplId])) {
773        if let Some(impls) = self.map.get(&trait_) {
774            callback(&impls.blanket_impls);
775            for impls in impls.non_blanket_impls.values() {
776                callback(impls);
777            }
778        }
779    }
780
781    pub fn for_self_ty(&self, self_ty: &SimplifiedType, mut callback: impl FnMut(&[ImplId])) {
782        for for_trait in self.map.values() {
783            if let Some(for_ty) = for_trait.non_blanket_impls.get(self_ty) {
784                callback(for_ty);
785            }
786        }
787    }
788
789    pub fn for_each_crate_and_block(
790        db: &dyn HirDatabase,
791        krate: Crate,
792        block: Option<BlockId>,
793        for_each: &mut dyn FnMut(&TraitImpls),
794    ) {
795        let blocks = std::iter::successors(block, |block| block.loc(db).module.containing_block());
796        blocks.filter_map(|block| Self::for_block(db, block).as_deref()).for_each(&mut *for_each);
797        Self::for_crate_and_deps(db, krate).iter().map(|it| &**it).for_each(for_each);
798    }
799
800    /// Like [`Self::for_each_crate_and_block()`], but takes in account two blocks, one for a trait and one for a self type.
801    pub fn for_each_crate_and_block_trait_and_type(
802        db: &dyn HirDatabase,
803        krate: Crate,
804        type_block: Option<BlockId>,
805        trait_block: Option<BlockId>,
806        for_each: &mut dyn FnMut(&TraitImpls),
807    ) {
808        let in_self_and_deps = TraitImpls::for_crate_and_deps(db, krate);
809        in_self_and_deps.iter().for_each(|impls| for_each(impls));
810
811        // We must not provide duplicate impls to the solver. Therefore we work with the following strategy:
812        // start from each block, and walk ancestors until you meet the other block. If they never meet,
813        // that means there can't be duplicate impls; if they meet, we stop the search of the deeper block.
814        // This breaks when they are equal (both will stop immediately), therefore we handle this case
815        // specifically.
816        let blocks_iter = |block: Option<BlockId>| {
817            std::iter::successors(block, |block| block.loc(db).module.containing_block())
818        };
819        let for_each_block = |current_block: Option<BlockId>, other_block: Option<BlockId>| {
820            blocks_iter(current_block)
821                .take_while(move |&block| {
822                    other_block.is_none_or(|other_block| other_block != block)
823                })
824                .filter_map(move |block| TraitImpls::for_block(db, block).as_deref())
825        };
826        if trait_block == type_block {
827            blocks_iter(trait_block)
828                .filter_map(|block| TraitImpls::for_block(db, block).as_deref())
829                .for_each(for_each);
830        } else {
831            for_each_block(trait_block, type_block).for_each(&mut *for_each);
832            for_each_block(type_block, trait_block).for_each(for_each);
833        }
834    }
