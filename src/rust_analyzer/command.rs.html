18/// Cargo output is structured as one JSON per line. This trait abstracts parsing one line of
19/// cargo output into a Rust data type
20pub(crate) trait CargoParser<T>: Send + 'static {
21    fn from_line(&self, line: &str, error: &mut String) -> Option<T>;
22    fn from_eof(&self) -> Option<T>;
23}
24
25struct CargoActor<T> {
26    parser: Box<dyn CargoParser<T>>,
27    sender: Sender<T>,
28    stdout: ChildStdout,
29    stderr: ChildStderr,
30}
31
32impl<T: Sized + Send + 'static> CargoActor<T> {
33    fn new(
34        parser: impl CargoParser<T>,
35        sender: Sender<T>,
36        stdout: ChildStdout,
37        stderr: ChildStderr,
38    ) -> Self {
39        let parser = Box::new(parser);
40        CargoActor { parser, sender, stdout, stderr }
41    }
42}
43
44impl<T: Sized + Send + 'static> CargoActor<T> {
45    fn run(self, outfile: Option<Utf8PathBuf>) -> io::Result<(bool, String)> {
46        // We manually read a line at a time, instead of using serde's
47        // stream deserializers, because the deserializer cannot recover
48        // from an error, resulting in it getting stuck, because we try to
49        // be resilient against failures.
50        //
51        // Because cargo only outputs one JSON object per line, we can
52        // simply skip a line if it doesn't parse, which just ignores any
53        // erroneous output.
54
55        let mut stdout = outfile.as_ref().and_then(|path| {
56            _ = std::fs::create_dir_all(path);
57            Some(BufWriter::new(std::fs::File::create(path.join("stdout")).ok()?))
58        });
59        let mut stderr = outfile.as_ref().and_then(|path| {
60            _ = std::fs::create_dir_all(path);
61            Some(BufWriter::new(std::fs::File::create(path.join("stderr")).ok()?))
62        });
63
64        let mut stdout_errors = String::new();
65        let mut stderr_errors = String::new();
66        let mut read_at_least_one_stdout_message = false;
67        let mut read_at_least_one_stderr_message = false;
68        let process_line = |line: &str, error: &mut String| {
69            // Try to deserialize a message from Cargo or Rustc.
70            if let Some(t) = self.parser.from_line(line, error) {
71                self.sender.send(t).unwrap();
72                true
73            } else {
74                false
75            }
76        };
77        let output = streaming_output(
78            self.stdout,
79            self.stderr,
80            &mut |line| {
81                if let Some(stdout) = &mut stdout {
82                    _ = stdout.write_all(line.as_bytes());
83                    _ = stdout.write_all(b"\n");
84                }
85                if process_line(line, &mut stdout_errors) {
86                    read_at_least_one_stdout_message = true;
87                }
88            },
89            &mut |line| {
90                if let Some(stderr) = &mut stderr {
91                    _ = stderr.write_all(line.as_bytes());
92                    _ = stderr.write_all(b"\n");
93                }
94                if process_line(line, &mut stderr_errors) {
95                    read_at_least_one_stderr_message = true;
96                }
97            },
98            &mut || {
99                if let Some(t) = self.parser.from_eof() {
100                    self.sender.send(t).unwrap();
101                }
102            },
103        );
104
105        let read_at_least_one_message =
106            read_at_least_one_stdout_message || read_at_least_one_stderr_message;
107        let mut error = stdout_errors;
108        error.push_str(&stderr_errors);
109        match output {
110            Ok(_) => Ok((read_at_least_one_message, error)),
111            Err(e) => Err(io::Error::new(e.kind(), format!("{e:?}: {error}"))),
112        }
113    }
114}

116struct JodGroupChild(Box<dyn StdChildWrapper>);
117
118impl Drop for JodGroupChild {
119    fn drop(&mut self) {
120        _ = self.0.kill();
121        _ = self.0.wait();
122    }
123}

125/// A handle to a cargo process used for fly-checking.
126pub(crate) struct CommandHandle<T> {
127    /// The handle to the actual cargo process. As we cannot cancel directly from with
128    /// a read syscall dropping and therefore terminating the process is our best option.
129    child: JodGroupChild,
130    thread: stdx::thread::JoinHandle<io::Result<(bool, String)>>,
131    program: OsString,
132    arguments: Vec<OsString>,
133    current_dir: Option<PathBuf>,
134    _phantom: PhantomData<T>,
135}
136
137impl<T> fmt::Debug for CommandHandle<T> {
138    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
139        f.debug_struct("CommandHandle")
140            .field("program", &self.program)
141            .field("arguments", &self.arguments)
142            .field("current_dir", &self.current_dir)
143            .finish()
144    }
145}
146
147impl<T: Sized + Send + 'static> CommandHandle<T> {
148    pub(crate) fn spawn(
149        mut command: Command,
150        parser: impl CargoParser<T>,
151        sender: Sender<T>,
152        out_file: Option<Utf8PathBuf>,
153    ) -> std::io::Result<Self> {
154        command.stdout(Stdio::piped()).stderr(Stdio::piped()).stdin(Stdio::null());
155
156        let program = command.get_program().into();
157        let arguments = command.get_args().map(|arg| arg.into()).collect::<Vec<OsString>>();
158        let current_dir = command.get_current_dir().map(|arg| arg.to_path_buf());
159
160        let mut child = StdCommandWrap::from(command);
161        #[cfg(unix)]
162        child.wrap(process_wrap::std::ProcessSession);
163        #[cfg(windows)]
164        child.wrap(process_wrap::std::JobObject);
165        let mut child = child.spawn().map(JodGroupChild)?;
166
167        let stdout = child.0.stdout().take().unwrap();
168        let stderr = child.0.stderr().take().unwrap();
169
170        let actor = CargoActor::<T>::new(parser, sender, stdout, stderr);
171        let thread =
172            stdx::thread::Builder::new(stdx::thread::ThreadIntent::Worker, "CommandHandle")
173                .spawn(move || actor.run(out_file))
174                .expect("failed to spawn thread");
175        Ok(CommandHandle { program, arguments, current_dir, child, thread, _phantom: PhantomData })
176    }
177
178    pub(crate) fn cancel(mut self) {
179        let _ = self.child.0.kill();
180        let _ = self.child.0.wait();
181    }
182
183    pub(crate) fn join(mut self) -> io::Result<()> {
184        let exit_status = self.child.0.wait()?;
185        let (read_at_least_one_message, error) = self.thread.join()?;
186        if read_at_least_one_message || exit_status.success() {
187            Ok(())
188        } else {
189            Err(io::Error::other(format!(
190                "Cargo watcher failed, the command produced no valid metadata (exit code: {exit_status:?}):\n{error}"
191            )))
192        }
193    }
194}
