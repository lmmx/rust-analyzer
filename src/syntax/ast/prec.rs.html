11#[derive(Debug, Clone, Copy, PartialEq, PartialOrd)]
12pub enum ExprPrecedence {
13    // return val, break val, yield val, closures
14    Jump,
15    // = += -= *= /= %= &= |= ^= <<= >>=
16    Assign,
17    // .. ..=
18    Range,
19    // ||
20    LOr,
21    // &&
22    LAnd,
23    // == != < > <= >=
24    Compare,
25    // |
26    BitOr,
27    // ^
28    BitXor,
29    // &
30    BitAnd,
31    // << >>
32    Shift,
33    // + -
34    Sum,
35    // * / %
36    Product,
37    // as
38    Cast,
39    // unary - * ! & &mut
40    Prefix,
41    // function calls, array indexing, field expressions, method calls
42    Postfix,
43    // paths, loops,
44    Unambiguous,
45}
46
47impl ExprPrecedence {
48    pub fn needs_parentheses_in(self, other: ExprPrecedence) -> bool {
49        match other {
50            ExprPrecedence::Unambiguous => false,
51            // postfix ops have higher precedence than any other operator, so we need to wrap
52            // any inner expression that is below
53            ExprPrecedence::Postfix => self < ExprPrecedence::Postfix,
54            // We need to wrap all binary like things, thats everything below prefix except for
55            // jumps (as those are prefix operations as well)
56            ExprPrecedence::Prefix => ExprPrecedence::Jump < self && self < ExprPrecedence::Prefix,
57            parent => self <= parent,
58        }
59    }
60}

72pub fn precedence(expr: &ast::Expr) -> ExprPrecedence {
73    match expr {
74        Expr::ClosureExpr(closure) => match closure.ret_type() {
75            None => ExprPrecedence::Jump,
76            Some(_) => ExprPrecedence::Unambiguous,
77        },
78
79        Expr::BreakExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
80        Expr::BecomeExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
81        Expr::ReturnExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
82        Expr::YeetExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
83        Expr::YieldExpr(e) if e.expr().is_some() => ExprPrecedence::Jump,
84
85        Expr::BreakExpr(_)
86        | Expr::BecomeExpr(_)
87        | Expr::ReturnExpr(_)
88        | Expr::YeetExpr(_)
89        | Expr::YieldExpr(_)
90        | Expr::ContinueExpr(_) => ExprPrecedence::Unambiguous,
91
92        Expr::RangeExpr(..) => ExprPrecedence::Range,
93
94        Expr::BinExpr(bin_expr) => match bin_expr.op_kind() {
95            Some(it) => match it {
96                BinaryOp::LogicOp(logic_op) => match logic_op {
97                    ast::LogicOp::And => ExprPrecedence::LAnd,
98                    ast::LogicOp::Or => ExprPrecedence::LOr,
99                },
100                BinaryOp::ArithOp(arith_op) => match arith_op {
101                    ast::ArithOp::Add | ast::ArithOp::Sub => ExprPrecedence::Sum,
102                    ast::ArithOp::Div | ast::ArithOp::Rem | ast::ArithOp::Mul => {
103                        ExprPrecedence::Product
104                    }
105                    ast::ArithOp::Shl | ast::ArithOp::Shr => ExprPrecedence::Shift,
106                    ast::ArithOp::BitXor => ExprPrecedence::BitXor,
107                    ast::ArithOp::BitOr => ExprPrecedence::BitOr,
108                    ast::ArithOp::BitAnd => ExprPrecedence::BitAnd,
109                },
110                BinaryOp::CmpOp(_) => ExprPrecedence::Compare,
111                BinaryOp::Assignment { .. } => ExprPrecedence::Assign,
112            },
113            None => ExprPrecedence::Unambiguous,
114        },
115        Expr::CastExpr(_) => ExprPrecedence::Cast,
116
117        Expr::LetExpr(_) | Expr::PrefixExpr(_) | Expr::RefExpr(_) => ExprPrecedence::Prefix,
118
119        Expr::AwaitExpr(_)
120        | Expr::CallExpr(_)
121        | Expr::FieldExpr(_)
122        | Expr::IndexExpr(_)
123        | Expr::MethodCallExpr(_)
124        | Expr::TryExpr(_) => ExprPrecedence::Postfix,
125
126        Expr::ArrayExpr(_)
127        | Expr::AsmExpr(_)
128        | Expr::BlockExpr(_)
129        | Expr::ForExpr(_)
130        | Expr::FormatArgsExpr(_)
131        | Expr::IfExpr(_)
132        | Expr::Literal(_)
133        | Expr::LoopExpr(_)
134        | Expr::MacroExpr(_)
135        | Expr::MatchExpr(_)
136        | Expr::OffsetOfExpr(_)
137        | Expr::ParenExpr(_)
138        | Expr::PathExpr(_)
139        | Expr::RecordExpr(_)
140        | Expr::TupleExpr(_)
141        | Expr::UnderscoreExpr(_)
142        | Expr::WhileExpr(_) => ExprPrecedence::Unambiguous,
143    }
144}

146fn check_ancestry(ancestor: &SyntaxNode, descendent: &SyntaxNode) -> bool {
147    let bail = || always!(false, "{} is not an ancestor of {}", ancestor, descendent);
148
149    if !ancestor.text_range().contains_range(descendent.text_range()) {
150        return bail();
151    }
152
153    for anc in descendent.ancestors() {
154        if anc == *ancestor {
155            return true;
156        }
157    }
158
159    bail()
160}

173    /// Returns `true` if `self` would need to be wrapped in parentheses given that its parent is `parent`.
174    pub fn needs_parens_in(&self, parent: &SyntaxNode) -> bool {
175        self.needs_parens_in_place_of(parent, self.syntax())
176    }

178    /// Returns `true` if `self` would need to be wrapped in parentheses if it replaces `place_of`
179    /// given that `place_of`'s parent is `parent`.
180    pub fn needs_parens_in_place_of(&self, parent: &SyntaxNode, place_of: &SyntaxNode) -> bool {
181        if !check_ancestry(parent, place_of) {
182            return false;
183        }
184
185        match_ast! {
186            match parent {
187                ast::Expr(e) => self.needs_parens_in_expr(&e, place_of),
188                ast::Stmt(e) => self.needs_parens_in_stmt(Some(&e)),
189                ast::StmtList(_) => self.needs_parens_in_stmt(None),
190                ast::ArgList(_) => false,
191                ast::MatchArm(_) => false,
192                _ => false,
193            }
194        }
195    }

197    fn needs_parens_in_expr(&self, parent: &Expr, place_of: &SyntaxNode) -> bool {
198        // Parentheses are necessary when calling a function-like pointer that is a member of a struct or union
199        // (e.g. `(a.f)()`).
200        let is_parent_call_expr = matches!(parent, ast::Expr::CallExpr(_));
201        let is_field_expr = matches!(self, ast::Expr::FieldExpr(_));
202        if is_parent_call_expr && is_field_expr {
203            return true;
204        }
205
206        // Special-case block weirdness
207        if parent.child_is_followed_by_a_block() {
208            use Expr::*;
209            match self {
210                // Cases like `if return {}` (need parens or else `{}` is returned, instead of being `if`'s body)
211                ReturnExpr(e) if e.expr().is_none() => return true,
212                BreakExpr(e) if e.expr().is_none() => return true,
213                YieldExpr(e) if e.expr().is_none() => return true,
214
215                // Same but with `..{}`
216                RangeExpr(e) if matches!(e.end(), Some(BlockExpr(..))) => return true,
217
218                // Similarly with struct literals, e.g. `if S{} == 1 {}`
219                _ if self.contains_exterior_struct_lit() => return true,
220                _ => {}
221            }
222        }
223
224        // Special-case `return.f()`
225        if self.is_ret_like_with_no_value() && parent.is_postfix() {
226            return false;
227        }
228
229        // Special-case prefix operators with return/break/etc without value
230        // e.g., `!(return)` - parentheses are necessary
231        if self.is_ret_like_with_no_value() && parent.is_prefix() {
232            return true;
233        }
234
235        if self.is_paren_like()
236            || parent.is_paren_like()
237            || self.is_prefix()
238                && (parent.is_prefix()
239                    || !self.is_ordered_before_parent_in_place_of(parent, place_of))
240            || self.is_postfix()
241                && (parent.is_postfix()
242                    || self.is_ordered_before_parent_in_place_of(parent, place_of))
243        {
244            return false;
245        }
246
247        let (left, right, inv) = match self.is_ordered_before_parent_in_place_of(parent, place_of) {
248            true => (self, parent, false),
249            false => (parent, self, true),
250        };
251
252        let (_, left_right_bp) = left.binding_power();
253        let (right_left_bp, _) = right.binding_power();
254
255        (left_right_bp < right_left_bp) ^ inv
256    }

258    fn needs_parens_in_stmt(&self, stmt: Option<&ast::Stmt>) -> bool {
259        use Expr::*;
260
261        // Prevent false-positives in cases like `fn x() -> u8 { ({ 0 } + 1) }`,
262        // `{ { 0 } + 1 }` won't parse -- `{ 0 }` would be parsed as a self-contained stmt,
263        // leaving `+ 1` as a parse error.
264        let mut innermost = self.clone();
265        loop {
266            let next = match &innermost {
267                BinExpr(e) => e.lhs(),
268                CallExpr(e) => e.expr(),
269                CastExpr(e) => e.expr(),
270                IndexExpr(e) => e.base(),
271                _ => break,
272            };
273
274            if let Some(next) = next {
275                innermost = next;
276                if !innermost.requires_semi_to_be_stmt() {
277                    return true;
278                }
279            } else {
280                break;
281            }
282        }
283
284        // Not every expression can be followed by `else` in the `let-else`
285        if let Some(ast::Stmt::LetStmt(e)) = stmt
286            && e.let_else().is_some()
287        {
288            match self {
289                BinExpr(e)
290                    if e.op_kind()
291                        .map(|op| matches!(op, BinaryOp::LogicOp(_)))
292                        .unwrap_or(false) =>
293                {
294                    return true;
295                }
296                _ if self.clone().trailing_brace().is_some() => return true,
297                _ => {}
298            }
299        }
300
301        false
302    }

304    /// Returns left and right so-called "binding powers" of this expression.
305    fn binding_power(&self) -> (u8, u8) {
306        use ast::{ArithOp::*, BinaryOp::*, Expr::*, LogicOp::*};
307
308        match self {
309            // (0, 0)   -- paren-like/nullary
310            // (0, N)   -- prefix
311            // (N, 0)   -- postfix
312            // (N, N)   -- infix, requires parens
313            // (N, N+1) -- infix, left to right associative
314            // (N+1, N) -- infix, right to left associative
315            // N is odd
316            //
317            ContinueExpr(_) => (0, 0),
318
319            ClosureExpr(_) | ReturnExpr(_) | BecomeExpr(_) | YieldExpr(_) | YeetExpr(_)
320            | BreakExpr(_) | OffsetOfExpr(_) | FormatArgsExpr(_) | AsmExpr(_) => (0, 1),
321
322            RangeExpr(_) => (5, 5),
323
324            BinExpr(e) => {
325                // Return a dummy value if we don't know the op
326                let Some(op) = e.op_kind() else { return (0, 0) };
327                match op {
328                    Assignment { .. } => (4, 3),
329                    //
330                    // Ranges are here in order :)
331                    //
332                    LogicOp(op) => match op {
333                        Or => (7, 8),
334                        And => (9, 10),
335                    },
336                    CmpOp(_) => (11, 11),
337                    ArithOp(op) => match op {
338                        BitOr => (13, 14),
339                        BitXor => (15, 16),
340                        BitAnd => (17, 18),
341                        Shl | Shr => (19, 20),
342                        Add | Sub => (21, 22),
343                        Mul | Div | Rem => (23, 24),
344                    },
345                }
346            }
347
348            CastExpr(_) => (25, 26),
349
350            RefExpr(_) | LetExpr(_) | PrefixExpr(_) => (0, 27),
351
352            AwaitExpr(_) | CallExpr(_) | MethodCallExpr(_) | IndexExpr(_) | TryExpr(_)
353            | MacroExpr(_) => (29, 0),
354
355            FieldExpr(_) => (31, 32),
356
357            ArrayExpr(_) | TupleExpr(_) | Literal(_) | PathExpr(_) | ParenExpr(_) | IfExpr(_)
358            | WhileExpr(_) | ForExpr(_) | LoopExpr(_) | MatchExpr(_) | BlockExpr(_)
359            | RecordExpr(_) | UnderscoreExpr(_) => (0, 0),
360        }
361    }

363    fn is_paren_like(&self) -> bool {
364        matches!(self.binding_power(), (0, 0))
365    }

367    fn is_prefix(&self) -> bool {
368        matches!(self.binding_power(), (0, 1..))
369    }
370
371    fn is_postfix(&self) -> bool {
372        matches!(self.binding_power(), (1.., 0))
373    }

375    /// Returns `true` if this expression can't be a standalone statement.
376    fn requires_semi_to_be_stmt(&self) -> bool {
377        use Expr::*;
378        !matches!(
379            self,
380            IfExpr(..) | MatchExpr(..) | BlockExpr(..) | WhileExpr(..) | LoopExpr(..) | ForExpr(..)
381        )
382    }

384    /// If an expression ends with `}`, returns the innermost expression ending in this `}`.
385    fn trailing_brace(mut self) -> Option<Expr> {
386        use Expr::*;
387
388        loop {
389            let rhs = match self {
390                RefExpr(e) => e.expr(),
391                BinExpr(e) => e.rhs(),
392                BreakExpr(e) => e.expr(),
393                LetExpr(e) => e.expr(),
394                RangeExpr(e) => e.end(),
395                ReturnExpr(e) => e.expr(),
396                PrefixExpr(e) => e.expr(),
397                YieldExpr(e) => e.expr(),
398                ClosureExpr(e) => e.body(),
399
400                BlockExpr(..) | ForExpr(..) | IfExpr(..) | LoopExpr(..) | MatchExpr(..)
401                | RecordExpr(..) | WhileExpr(..) => break Some(self),
402                _ => break None,
403            };
404
405            self = rhs?;
406        }
407    }

409    /// Expressions that syntactically contain an "exterior" struct literal i.e., not surrounded by any
410    /// parens or other delimiters, e.g., `X { y: 1 }`, `X { y: 1 }.method()`, `foo == X { y: 1 }` and
411    /// `X { y: 1 } == foo` all do, but `(X { y: 1 }) == foo` does not.
412    fn contains_exterior_struct_lit(&self) -> bool {
413        return contains_exterior_struct_lit_inner(self).is_some();
414
415        fn contains_exterior_struct_lit_inner(expr: &Expr) -> Option<()> {
416            use Expr::*;
417
418            match expr {
419                RecordExpr(..) => Some(()),
420
421                // X { y: 1 } + X { y: 2 }
422                BinExpr(e) => e
423                    .lhs()
424                    .as_ref()
425                    .and_then(contains_exterior_struct_lit_inner)
426                    .or_else(|| e.rhs().as_ref().and_then(contains_exterior_struct_lit_inner)),
427
428                // `&X { y: 1 }`, `X { y: 1 }.y`, `X { y: 1 }.bar(...)`, etc
429                IndexExpr(e) => contains_exterior_struct_lit_inner(&e.base()?),
430                AwaitExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
431                PrefixExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
432                CastExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
433                FieldExpr(e) => contains_exterior_struct_lit_inner(&e.expr()?),
434                MethodCallExpr(e) => contains_exterior_struct_lit_inner(&e.receiver()?),
435
436                _ => None,
437            }
438        }
439    }

441    /// Returns true if self is one of `return`, `break`, `continue` or `yield` with **no associated value**.
442    pub fn is_ret_like_with_no_value(&self) -> bool {
443        use Expr::*;
444
445        match self {
446            ReturnExpr(e) => e.expr().is_none(),
447            BreakExpr(e) => e.expr().is_none(),
448            ContinueExpr(_) => true,
449            YieldExpr(e) => e.expr().is_none(),
450            BecomeExpr(e) => e.expr().is_none(),
451            _ => false,
452        }
453    }

455    fn is_ordered_before_parent_in_place_of(&self, parent: &Expr, place_of: &SyntaxNode) -> bool {
456        use Expr::*;
457        use rowan::TextSize;
458
459        let self_range = self.syntax().text_range();
460        let place_of_range = place_of.text_range();
461
462        let self_order_adjusted = order(self) - self_range.start() + place_of_range.start();
463
464        let parent_order = order(parent);
465        let parent_order_adjusted = if parent_order <= place_of_range.start() {
466            parent_order
467        } else if parent_order >= place_of_range.end() {
468            parent_order - place_of_range.len() + self_range.len()
469        } else {
470            return false;
471        };
472
473        return self_order_adjusted < parent_order_adjusted;
474
475        /// Returns text range that can be used to compare two expression for order (which goes first).
476        fn order(this: &Expr) -> TextSize {
477            // For non-paren-like operators: get the operator itself
478            let token = match this {
479                RangeExpr(e) => e.op_token(),
480                BinExpr(e) => e.op_token(),
481                CastExpr(e) => e.as_token(),
482                FieldExpr(e) => e.dot_token(),
483                AwaitExpr(e) => e.dot_token(),
484                BreakExpr(e) => e.break_token(),
485                CallExpr(e) => e.arg_list().and_then(|args| args.l_paren_token()),
486                ClosureExpr(e) => e.param_list().and_then(|params| params.l_paren_token()),
487                ContinueExpr(e) => e.continue_token(),
488                IndexExpr(e) => e.l_brack_token(),
489                MethodCallExpr(e) => e.dot_token(),
490                PrefixExpr(e) => e.op_token(),
491                RefExpr(e) => e.amp_token(),
492                ReturnExpr(e) => e.return_token(),
493                BecomeExpr(e) => e.become_token(),
494                TryExpr(e) => e.question_mark_token(),
495                YieldExpr(e) => e.yield_token(),
496                YeetExpr(e) => e.do_token(),
497                LetExpr(e) => e.let_token(),
498                OffsetOfExpr(e) => e.builtin_token(),
499                FormatArgsExpr(e) => e.builtin_token(),
500                AsmExpr(e) => e.builtin_token(),
501                ArrayExpr(_) | TupleExpr(_) | Literal(_) | PathExpr(_) | ParenExpr(_)
502                | IfExpr(_) | WhileExpr(_) | ForExpr(_) | LoopExpr(_) | MatchExpr(_)
503                | BlockExpr(_) | RecordExpr(_) | UnderscoreExpr(_) | MacroExpr(_) => None,
504            };
505
506            token.map(|t| t.text_range()).unwrap_or_else(|| this.syntax().text_range()).start()
507        }
508    }

510    fn child_is_followed_by_a_block(&self) -> bool {
511        use Expr::*;
512
513        match self {
514            ArrayExpr(_) | AwaitExpr(_) | BlockExpr(_) | CallExpr(_) | CastExpr(_)
515            | ClosureExpr(_) | FieldExpr(_) | IndexExpr(_) | Literal(_) | LoopExpr(_)
516            | MacroExpr(_) | MethodCallExpr(_) | ParenExpr(_) | PathExpr(_) | RecordExpr(_)
517            | TryExpr(_) | TupleExpr(_) | UnderscoreExpr(_) | OffsetOfExpr(_)
518            | FormatArgsExpr(_) | AsmExpr(_) => false,
519
520            // For BinExpr and RangeExpr this is technically wrong -- the child can be on the left...
521            BinExpr(_) | RangeExpr(_) | BreakExpr(_) | ContinueExpr(_) | PrefixExpr(_)
522            | RefExpr(_) | ReturnExpr(_) | BecomeExpr(_) | YieldExpr(_) | YeetExpr(_)
523            | LetExpr(_) => self
524                .syntax()
525                .parent()
526                .and_then(Expr::cast)
527                .map(|e| e.child_is_followed_by_a_block())
528                .unwrap_or(false),
529
530            ForExpr(_) | IfExpr(_) | MatchExpr(_) | WhileExpr(_) => true,
531        }
532    }
